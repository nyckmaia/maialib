91 float computeStandardDeviation(const std::vector<T>& v) const {
92 const float sum = std::accumulate(v.begin(), v.end(), 0);
93 const float mean = sum / v.size();
94
95 std::vector<float> diff(v.size());
96 std::transform(v.begin(), v.end(), diff.begin(), [mean](double x) { return x - mean; });
97 const float sq_sum = std::inner_product(diff.begin(), diff.end(), diff.begin(), 0.0);
98 const float stdev = std::sqrt(sq_sum / v.size());
99
100 return stdev;
101 }
